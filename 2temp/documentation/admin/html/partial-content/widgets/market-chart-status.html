var piedata = [
    { label: "Apple", data: [[1,40]], color: '#906094'},
    { label: "Android", data: [[1,20]], color: '#8CC152'},
    { label: "Windows Phone", data: [[1,50]], color: '#F6BB42'}
];

function labelFormatter(label, series) {
    return "<div style='font-size:8pt; text-align:center; padding:2px; color:white;'>" + label +
    "<br/>" + Math.round(series.percent) + "%</div>";
}

$.plot('#market-status-chart', piedata, {
    series: {
        pie: {
            show: true,
            radius: 500,
            label: {
                show: true,
                radius: 2/3,
                formatter: labelFormatter,
                threshold: 0.1
            }
        }
    },
    legend: {
        show: false
    },
    grid: {
        hoverable: true,
        clickable: true
    }
});
