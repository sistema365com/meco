$(window).load(function(){
                var container = document.querySelector('#blog-list');
                var msnry = new Masonry( container, {
                    // options
                    columnWidth: '.col-xs-6',
                    itemSelector: '.col-xs-6',
                    isAnimated: true
                });

                // CHECK ON LOAD
                if($(window).width() <= 480 )
                    msnry.destroy();

                // CHECK ON RESIZE
                $(window).resize(function(){
                    if($(this).width() <= 480 )
                        msnry.destroy();
                });
            });
